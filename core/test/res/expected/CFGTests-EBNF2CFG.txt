V = {<arguments>, <assignment>, <body>, <call_arguments>, <expression>, <factor>, <function>, <function_call>, <if>, <program>, <return>, <statement>, <term>}
T = {!, (, ), *, +, ,, -, /, ;, <, =, >, else, fn, identifier, if, let, literal, return, {, }}
P = {
  <arguments> ::= identifier|identifier , <arguments>
  <assignment> ::= identifier = <expression> ;|let identifier = <expression> ;
  <body> ::= <statement>|<statement> <body>
  <call_arguments> ::= <expression>|<expression> , <call_arguments>
  <expression> ::= <term>|<term> ! = <expression>|<term> + <expression>|<term> - <expression>|<term> < <expression>|<term> < = <expression>|<term> = = <expression>|<term> > <expression>|<term> > = <expression>
  <factor> ::= ! <expression>|( <expression> )|identifier|literal
  <function> ::= fn identifier ( ) { <body> }|fn identifier ( <arguments> ) { <body> }
  <function_call> ::= identifier ( ) ;|identifier ( <call_arguments> ) ;
  <if> ::= if <expression> { <body> }|if <expression> { <body> } else { <body> }
  <program> ::= <function>|<function> <program>
  <return> ::= return <expression> ;
  <statement> ::= <assignment>|<function_call>|<if>|<return>
  <term> ::= <factor>|<factor> * <term>|<factor> / <term>
}
